RDA - Relative Abundance by Oyster Density

library(ggrepel) 

comm_raw <- biobox %>%
  group_by(bay, site, species) %>%
  summarise(abundance = sum(number_individuals), .groups = "drop") %>%
  pivot_wider(
    names_from = species,
    values_from = abundance,
    values_fill = 0
  )
 
# Save site identifiers for later reattachment
site_ids <- comm_raw %>%
  select(bay, site)

# Remove non-numeric columns so matrix contains ONLY species data
comm_mat <- comm_raw %>%
  select(-bay, -site)

# Hellinger transform data:
# - Converts counts to relative abundances
# - Square-rooted to reduce dominance of abundant taxa
# - Makes data suitable for Euclidean-based methods like RDA
comm_hel <- decostand(comm_mat, method = "hellinger")


env <- hydro %>%
  group_by(bay, site) %>%
  summarise(
    temp = mean(water_temp_C, na.rm = TRUE),
    salinity = mean(sal_ppt, na.rm = TRUE),
    DO = mean(DO_mgL, na.rm = TRUE),
    pH = mean(pH, na.rm = TRUE),
    .groups = "drop"
  )

oysters_site <- oysters %>%
  filter(extraction_type == "Dredge", oyster_mortality == "live") %>%
  group_by(bay, site) %>%
  summarise(
    oyster_total = sum(oyster_total_count, na.rm = TRUE),
    .groups = "drop"
  )

#join datasets
env_full <- site_ids %>%
  left_join(oysters_site, by = c("bay", "site")) %>%
  left_join(env, by = c("bay", "site"))

# Constrained ordination:
# Species composition ~ environmental gradients
rda_mod <- rda(
  comm_hel ~ salinity + temp + DO + pH + oyster_total,
  data = env_full
)

# Displays:
# - Eigenvalues (variance explained)
# - Species–environment correlations
summary(rda_mod)

# Site scores:
# - Positions of each site in constrained ordination space
site_scores <- scores(rda_mod, display = "sites", scaling = 2)


# Environmental (biplot) scores:
# - These are the "loadings" of environmental variables
# - Arrow length = strength of correlation with ordination axes
env_scores  <- scores(rda_mod, display = "bp", scaling = 2)


# Species scores:
# - Location of species optima along gradients
species_scores <- scores(rda_mod, display = "species", scaling = 2)


plot_df <- data.frame(
  site_scores,
  oyster_density = factor(
    cut(
      env_full$oyster_total,
      breaks = c(-Inf, 50, Inf),
      labels = c("Low", "High")
    ),
    levels = c("Low", "High")
  )
)


# Identify species with strong contributions to RDA1–RDA2
# (distance from origin in ordination space)
species_strength <- rowSums(species_scores[,1:2]^2)

# Keep top 25% most influential species
top_species <- species_scores[species_strength > quantile(species_strength, 0.75), ]

# Prepare environmental vectors dataframe
env_df <- as.data.frame(env_scores)
env_df$var <- rownames(env_df)

RDA.plot<- ggplot(plot_df, aes(RDA1, RDA2, color = oyster_density)) +

  # Sites
  geom_point(size = 3.5, alpha = 0.9) +

  # Environmental vectors
  geom_segment(
    data = env_df,
    aes(x = 0, y = 0,
        xend = RDA1,
        yend = RDA2),
    inherit.aes = FALSE,
    arrow = arrow(length = unit(0.3, "cm")),
    linewidth = 0.8,
    color = "black"
  ) +

  # Vector labels – pushed to arrow tips and repelled
  geom_text_repel(
    data = env_df,
    aes(x = RDA1 * 1.25, y = RDA2 * 1.25, label = var),
    inherit.aes = FALSE,
    fontface = "bold",
    size = 4,
    color = "black",
    segment.color = NA
  ) +

  # Species (still subtle)
  geom_text(
    data = as.data.frame(top_species),
    aes(RDA1, RDA2, label = rownames(top_species)),
    inherit.aes = FALSE,
    size = 2.7,
    color = "grey40",
    alpha = 0.7
  ) +

  scale_color_manual(
    values = c(
      "Low" = "#99c1de",
      "High" = "#386fa4"
    )
  ) +

  labs(
    x = "RDA1",
    y = "RDA2",
    color = "Live oyster counts"
  ) +

  coord_equal() +
  theme_classic(base_size = 15)
